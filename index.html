Скрипт 2 (LocalScript в StarterGui)
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

local remote = ReplicatedStorage:WaitForChild("AdminPanelEvent")

local function showAd(message)
    local gui = LocalPlayer:WaitForChild("PlayerGui")
    local adGui = gui:FindFirstChild("AdMessageGui")
    if not adGui then
        adGui = Instance.new("ScreenGui")
        adGui.Name = "AdMessageGui"
        adGui.ResetOnSpawn = false
        adGui.Parent = gui
    end
    local label = adGui:FindFirstChild("AdLabel")
    if not label then
        label = Instance.new("TextLabel")
        label.Name = "AdLabel"
        label.Size = UDim2.new(0.7, 0, 0, 60)
        label.Position = UDim2.new(0.15, 0, 0, 20)
        label.BackgroundColor3 = Color3.fromRGB(0,0,0)
        label.BackgroundTransparency = 0.3
        label.TextColor3 = Color3.fromRGB(255,255,0)
        label.Font = Enum.Font.GothamBold
        label.TextSize = 32
        label.TextStrokeTransparency = 0.3
        label.TextStrokeColor3 = Color3.fromRGB(0,0,0)
        label.TextWrapped = true
        label.Parent = adGui
    end
    label.Text = message
    label.Visible = true
    task.wait(8)
    label.Visible = false
end

remote.OnClientEvent:Connect(function(data)
    if data and data.action == "broadcast" and data.message then
        showAd(data.message)
    end
end)
Скрипт 3
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players = game:GetService("Players")

local ADMIN_USERNAME = "BodrosikXX"

local remote = ReplicatedStorage:WaitForChild("AdminPanelEvent")

remote.OnServerEvent:Connect(function(player, data)
    if not data or type(data) ~= "table" then return end
    if player.Name ~= ADMIN_USERNAME then return end

    if data.action == "broadcast" and data.message then
        for _, plr in Players:GetPlayers() do
            remote:FireClient(plr, {action="broadcast", message=data.message})
        end
    elseif data.action == "give" and data.player and data.amount then
        for _, plr in Players:GetPlayers() do
            if plr.Name == data.player then
                local leaderstats = plr:FindFirstChild("leaderstats")
                if leaderstats then
                    local balance = leaderstats:FindFirstChild("Balance")
                    if balance then
                        balance.Value = balance.Value + data.amount
                    end
                end
            end
        end
    end
end)
